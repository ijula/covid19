today="$(date +%m-%d-%Y)"; grep Romania *.csv | awk -F',' '{print substr($1, 1, 10) " " $4 " " $5 " " $6}' | gnuplot -e "set terminal pngcairo enhanced background rgb 'black'; set title 'COVID-19' tc rgb 'green'; set key left top tc rgb 'green'; set xdata time; set timefmt \"%m-%d-%Y\"; set xrange ['03-01-2020':'$today']; show xrange; set border lc rgb 'green'; plot '-' u 1:2 t 'New Cases' w lines;" > chart2.png
